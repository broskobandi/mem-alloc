164static inline void *use_free_list(ptr_t **free_tail) {
165 /* Init ptr */
166 ptr_t *ptr = *free_tail;
167 ptr->is_valid = true;
168
169 /* Update free list. */
170 if ((*free_tail)->prev_free) {
171 *free_tail = (*free_tail)->prev_free;
172 (*free_tail)->next_free = NULL;
173 } else {
174 *free_tail = NULL;
175 }
176
177 ptr->next_free = NULL;
178 ptr->prev_free = NULL;
179
180 return ptr->mem;
181}

183static inline void *use_arena(size_t total_size, arena_t *arena) {
184 /* Init ptr */
185 ptr_t *ptr = (ptr_t*)&arena->buff[arena->offset];
186 arena->offset += total_size;
187 ptr->next_in_arena = NULL;
188 ptr->next_free = NULL;
189 ptr->prev_free = NULL;
190 ptr->is_valid = true;
191 ptr->is_mmap = false;
192 ptr->total_size = total_size;
193 ptr->mem = (void*)((unsigned char*)ptr + MEM_OFFSET);
194
195 /* Update ptr list. */
196 if (arena->ptrs_tail) {
197 arena->ptrs_tail->next_in_arena = ptr;
198 ptr->prev_in_arena = arena->ptrs_tail;
199 }
200 arena->ptrs_tail = ptr;
201 return ptr->mem;
202}

204static inline void *use_mmap(size_t total_size) {
205 /* Init ptr */
206 size_t updated_total_size = ROUNDUP(total_size, (size_t)getpagesize());
207 ptr_t *ptr = (ptr_t*)MMAP(updated_total_size);
208 if (!ptr) return NULL;
209 ptr->prev_in_arena = NULL;
210 ptr->next_in_arena = NULL;
211 ptr->next_free = NULL;
212 ptr->prev_free = NULL;
213 ptr->is_valid = true;
214 ptr->is_mmap = true;
215 ptr->total_size = updated_total_size;
216 ptr->mem = (void*)((unsigned char*)ptr + MEM_OFFSET);
217 return ptr->mem;
218}

220static inline void add_to_free_list(ptr_t *ptr, arena_t *arena) {
221 ptr->is_valid = false;
222 ptr_t **tail = &arena->free_ptr_tails[SIZE_CLASS(ptr->total_size)];
223 if (*tail) {
224 (*tail)->next_free = ptr;
225 ptr->prev_free = (*tail);
226 }
227 *tail = ptr;
228}

230static inline void remove_from_free_list(ptr_t *ptr, arena_t *arena) {
231 ptr_t **tail = &arena->free_ptr_tails[SIZE_CLASS(ptr->total_size)];
232 if (ptr->next_free) {
233 ptr->next_free->prev_free = ptr->prev_free;
234 } else {
235 *tail = ptr->prev_free;
236 }
237 if (ptr->prev_free)
238 ptr->prev_free->next_free = ptr->next_free;
239}

241static inline void merge_neighbouring_ptrs(ptr_t *ptr, arena_t *arena) {
242 if (ptr->next_in_arena && !ptr->next_in_arena->is_valid) {
243 remove_from_free_list(ptr->next_in_arena, arena);
244 ptr->total_size += ptr->next_in_arena->total_size;
245 ptr->next_in_arena = ptr->next_in_arena->next_in_arena;
246 ptr->next_in_arena->prev_in_arena = ptr;
247 remove_from_free_list(ptr, arena);
248 add_to_free_list(ptr, arena);
249 }
250 if (ptr->prev_in_arena && !ptr->prev_in_arena->is_valid) {
251 remove_from_free_list(ptr, arena);
252 ptr->prev_in_arena->total_size += ptr->total_size;
253 ptr->prev_in_arena->next_in_arena = ptr->next_in_arena;
254 if (ptr->next_in_arena)
255 ptr->next_in_arena->prev_in_arena = ptr->prev_in_arena;
256 remove_from_free_list(ptr->prev_in_arena, arena);
257 add_to_free_list(ptr->prev_in_arena, arena);
258 }

261static inline void move_ptr(ptr_t **ptr, size_t size) {
262 ptr_t *new_ptr = (ptr_t*)((unsigned char*)*ptr + size);
263 memcpy(new_ptr, *ptr, sizeof(ptr_t));
264 *ptr = new_ptr;
265 (*ptr)->mem = ((unsigned char*)*ptr + MEM_OFFSET);
266 if ((*ptr)->next_in_arena)
267 (*ptr)->next_in_arena->prev_in_arena = *ptr;
268 if ((*ptr)->prev_in_arena)
269 (*ptr)->prev_in_arena->next_in_arena = *ptr;
270 if ((*ptr)->next_free)
271 (*ptr)->next_free->prev_free = *ptr;
272 if ((*ptr)->prev_free)
273 (*ptr)->prev_free->next_free = *ptr;
274}

276static inline void *realloc_in_place(
277 ptr_t *ptr, size_t total_size, arena_t *arena
278) {
279 /* If current size is enough */
280 if (ptr->total_size >= total_size)
281 return ptr->mem;
282
283 ptr_t **next = &ptr->next_in_arena;
284 size_t extra_size_needed =
285 ROUNDUP(total_size - ptr->total_size, MIN_ALLOC_SIZE);
286
287 /* If next pointer is free and is bigger than needed */
288 if (
289 *next && !(*next)->is_valid &&
290 (*next)->total_size >=
291 extra_size_needed + MEM_OFFSET + MIN_ALLOC_SIZE
292 ) {
293 move_ptr(next, extra_size_needed);
294 (*next)->total_size -= extra_size_needed;
295 remove_from_free_list(*next, arena);
296 add_to_free_list(*next, arena);
297 ptr->total_size += extra_size_needed;
298 return ptr->mem;
299 }
300
301 /* If next pointer is free and is just as big as needed */
302 if (
303 *next && !(*next)->is_valid &&
304 (*next)->total_size == extra_size_needed
305 ) {
306 remove_from_free_list(*next, arena);
307 ptr->total_size += extra_size_needed;
308 arena->offset += extra_size_needed;
309 return ptr->mem;
310 }
311
312 /* If there is no next pointer. */
313 if (!*next && arena->offset + extra_size_needed <= ARENA_SIZE) {
314 ptr->total_size += extra_size_needed;
315 arena->offset += extra_size_needed;
316 return ptr->mem;
317 }
318
319 return NULL;
320}
